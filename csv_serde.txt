CREATE TABLE csv_table (
  name STRING,
  location STRING
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.OpenCSVSerde'
WITH SERDEPROPERTIES (
  'separatorChar' = ',',
  'quoteChar' = '\"',
  'escapeChar' = '\\'
)
STORED AS TEXTFILE
TBLPROPERTIES (
  'skip.header.line.count' = '1'
);
